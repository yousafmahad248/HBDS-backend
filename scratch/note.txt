require('dotenv').config();
const { createNotification } = require('./controllers/notificationController');
const mongoose = require('mongoose');

async function trigger() {
    await mongoose.connect(process.env.MONGODB_URI);
    console.log("Connected to DB");

    console.log("Triggering notification for test-user-id...");
    await createNotification({
        recipient: new mongoose.Types.ObjectId(), // dummy
        recipientModel: 'User',
        title: 'Socket Test',
        message: 'This is a real-time test.',
        type: 'SYSTEM'
    });
    
    // Wait for the global.io to be initialized in server.js... 
    // Actually, this script won't have global.io because it's not running through server.js.
    // I need to run a script that connects to the RUNNING server.
    
    mongoose.disconnect();
}
